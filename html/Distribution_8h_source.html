114 
115  for(uint64_t i = 1; i<n; i++){
116  target.pdf_[i] = w1*dst1.pdf_[i] + w2*dst2.pdf_[i];
117  target.cdf_[i] = target.cdf_[i-1] + target.pdf_[i];
